function blendHarmonicSignatures(signatures, weights) {
  // Weighted average in HSL color space
  const totalWeight = weights.reduce((a, b) => a + b, 0)
  
  const blendedHue = signatures.reduce((sum, sig, i) => 
    sum + (sig.baseHue * weights[i]), 0) / totalWeight
  
  const blendedAccent = signatures.reduce((sum, sig, i) => 
    sum + (sig.accentHue * weights[i]), 0) / totalWeight
  
  const blendedIntensity = signatures.reduce((sum, sig, i) => 
    sum + (sig.intensity * weights[i]), 0) / totalWeight
  
  const blendedWarmth = signatures.reduce((sum, sig, i) => 
    sum + (sig.warmth * weights[i]), 0) / totalWeight
  
  return {
    name: `Blended: ${signatures.map(s => s.name).join(' + ')}`,
    baseHue: blendedHue,
    accentHue: blendedAccent,
    intensity: blendedIntensity,
